1class Solution {
2public:
3    /**
4     * Calculates the score for each index i in the given array nums.
5     * The score of an index i is the number of indices j such that:
6     * i < j < n and nums[i], nums[j] have different parity.
7     * 
8     * Time Complexity: O(n)
9     * Space Complexity: O(n) for the output array.
10     */
11    vector<int> countOppositeParity(vector<int>& nums) {
12        int n = nums.size();
13        vector<int> answer(n);
14        int even_count = 0;
15        int odd_count = 0;
16        
17        // Traverse the array backwards to pre-calculate the counts of even and odd
18        // numbers that appear to the right of each index i.
19        for (int i = n - 1; i >= 0; --i) {
20            // Check if nums[i] is even. Since 1 <= nums[i] <= 100, this is sufficient.
21            if (nums[i] % 2 == 0) {
22                // If nums[i] is even, its score is the number of odd elements to its right.
23                answer[i] = odd_count;
24                // Increment even_count for elements to the left.
25                even_count++;
26            } else {
27                // If nums[i] is odd, its score is the number of even elements to its right.
28                answer[i] = even_count;
29                // Increment odd_count for elements to the left.
30                odd_count++;
31            }
32        }
33        
34        return answer;
35    }
36};
