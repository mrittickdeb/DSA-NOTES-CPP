1class Solution {
2public:
3    vector<int> countOppositeParity(vector<int>& nums) {
4        int n = nums.size();
5        vector<int> answer(n);
6        int odd_count = 0;
7        int even_count = 0;
8        
9        // Traverse the array from right to left to count the suffix parities.
10        for (int i = n - 1; i >= 0; --i) {
11            if (nums[i] % 2 == 0) {
12                // For an even number, its score is the count of odd numbers to its right.
13                answer[i] = odd_count;
14                even_count++;
15            } else {
16                // For an odd number, its score is the count of even numbers to its right.
17                answer[i] = even_count;
18                odd_count++;
19            }
20        }
21        
22        return answer;
23    }
24};
